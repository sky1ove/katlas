def smoothsegment(seg, Nsmooth=100):
    """Smooths the segment for better visualization in polar coordinates."""
    return np.concatenate([[seg[0]], np.linspace(seg[1], seg[2], Nsmooth), [seg[3]]])

def plot_dendrogram(icoord, dcoord, labels, figsize, polar=True):
    """Plots a dendrogram in polar coordinates with labels properly oriented around the circle.
    
    Args:
        icoord (ndarray): Array of x coordinates for the dendrogram.
        dcoord (ndarray): Array of y coordinates for the dendrogram.
        labels (list): List of labels corresponding to the dendrogram leaves.
        figsize (tuple): Figure size.
        polar (bool): If True, plots in polar coordinates.
    """
    dcoord = -np.log(dcoord + 1)
    imax = icoord.max()
    imin = icoord.min()
    icoord = (icoord - imin) / (imax - imin) * 2 * np.pi
    
    with plt.style.context("seaborn-white"):
        fig = plt.figure(figsize=figsize)
        ax = fig.add_subplot(111, polar=polar)
        
        # Plot each segment
        for xs, ys in zip(icoord, dcoord):
            xs = smoothsegment(xs)
            ys = smoothsegment(ys)
            ax.plot(xs, ys, color="black")
        
        # Remove polar grid and labels
        ax.grid(False)
        ax.spines['polar'].set_visible(False)
        # ax.set_xticks(icoord)  # Set ticks at label positions
        
        Nxticks = len(labels)
        xticks = np.linspace(0, 2 * np.pi, Nxticks, endpoint=False)
        ax.set_xticks(xticks)

def get_newick(node, parent_dist, leaf_names, newick='') -> str:
    """
    Convert sciply.cluster.hierarchy.to_tree()-output to Newick format.
    Referenced from: https://stackoverflow.com/questions/28222179/save-dendrogram-to-newick-format
    
    :param node: output of sciply.cluster.hierarchy.to_tree()
    :param parent_dist: output of sciply.cluster.hierarchy.to_tree().dist
    :param leaf_names: list of leaf names
    :param newick: leave empty, this variable is used in recursion.
    :returns: tree in Newick format
    
    """
    if node.is_leaf():
        return "%s:%.2f%s" % (leaf_names[node.id], parent_dist - node.dist, newick)
    else:
        if len(newick) > 0:
            newick = "):%.2f%s" % (parent_dist - node.dist, newick)
        else:
            newick = ");"
        newick = get_newick(node.get_left(), node.dist, leaf_names, newick=newick)
        newick = get_newick(node.get_right(), node.dist, leaf_names, newick=",%s" % (newick))
        newick = "(%s" % (newick)
        return newick
